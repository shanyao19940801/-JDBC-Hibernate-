 #返回到某个节点，不保留修改。
git reset --hard HASH
#返回到某个节点。保留修改
git reset --soft HASH

#从远程下载最新的内容，而不尝试合并或者转换任何内容。
git fetch --all
#然后 git reset 将主分支重置为你刚刚获取的内容。 --hard 选项将更改工作树中的所有文件以匹配 origin/master 中的文件
git reset --hard origin/master
